Iniciando teste T

Author

Yuri

Importar dados

library(tidyverse)
library(readxl)
mg <- read_excel("dados-diversos.xlsx")
mg |> 
  ggplot(aes(trat, comp))+
  geom_jitter(width =0.05)+
  geom_boxplot(fill = NA, 
               outlier.colour = NA)+
  ylim(5, 20)+
  annotate(geom = "text",
           x = 0.7, y = 19,
           label = "t = 8.12; P < 0.001;")

mg |> 
  
  ggplot(aes(trat, comp))+
  stat_summary(fun.data = "mean_se")

mg2 <- mg |> 
  pivot_wider(1, 
              names_from = trat,
              values_from = comp)
t <- t.test(mg2$Mg2, mg2$control)
