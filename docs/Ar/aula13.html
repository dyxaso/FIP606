library(tidyverse)
library(readxl)

estande <- read_excel("dados-diversos.xlsx", "estande")

estande |> 
  ggplot(aes(trat, nplants))+
  geom_point()+
  facet_wrap(~exp)+
  ylim(0, max(estande$nplants))+
  geom_smooth(se =  FALSE, method = "lm")
